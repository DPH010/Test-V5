    <script>

        // Array of quiz questions, answers, and explanations

        const questions = [

                answers: [

                    { text: "Moana", correct: false }

                ],

                explanation: "'Spirited Away' is a celebrated Japanese animated fantasy film directed by Hayao Miyazaki, produced by Studio Ghibli."

            },

            {

                question: "What year was the first 'Star Wars' movie (A New Hope) released?",

                answers: [

                    { text: "1975", correct: false },

                    { text: "1980", correct: false },

                    { text: "1977", correct: true },

                    { text: "1983", correct: false }

                ],

                explanation: "The original 'Star Wars: A New Hope' premiered in 1977, changing cinematic history forever."

            }

        ];


        // Get DOM elements

        const questionDisplay = document.getElementById('question-display');

        const answerButtons = document.getElementById('answer-buttons');

        const feedbackArea = document.getElementById('feedback-area');

        const feedbackMessage = document.getElementById('feedback-message');

        const nextButton = document.getElementById('next-button');

        const learnMoreButton = document.getElementById('learn-more-button');

        const resultsArea = document.getElementById('results-area');

        const scoreDisplay = document.getElementById('score-display');

        const restartButton = document.getElementById('restart-button');

        const generateNewQuestionButton = document.getElementById('generate-new-question-button');


        let currentQuestionIndex = 0;

        let score = 0;

        let quizActive = true; // State to control if quiz is in progress


        /**

         * Initializes the quiz or restarts it to the beginning state.

         */

        function startQuiz() {

            currentQuestionIndex = 0;

            score = 0;

            quizActive = true;

            // Hide results and feedback, show quiz elements

            resultsArea.classList.add('hidden');

            feedbackArea.classList.add('hidden');

            nextButton.classList.add('hidden'); // Hide next button until answer is selected

            learnMoreButton.classList.add('hidden'); // Hide learn more button

            showQuestion();

        }


        /**

         * Displays the current question and its answer options.

         */

        function showQuestion() {

            resetState(); // Clear previous answers and feedback

            const questionData = questions[currentQuestionIndex];

            questionDisplay.textContent = questionData.question; // Display question text


            // Create buttons for each answer option

            questionData.answers.forEach(answer => {

                const button = document.createElement('button');

                button.textContent = answer.text;

                button.classList.add('answer-button', 'rounded-xl', 'px-4', 'py-3', 'text-lg', 'font-medium', 'text-left');

                // Store correct status as a data attribute

                if (answer.correct) {

                    button.dataset.correct = true;

                }

                button.addEventListener('click', selectAnswer); // Add click listener

                answerButtons.appendChild(button);

            });

        }


        /**

         * Resets the UI state for a new question: clears buttons, hides feedback and next button.

         */

        function resetState() {

            // Remove all existing answer buttons

            while (answerButtons.firstChild) {

                answerButtons.removeChild(answerButtons.firstChild);

            }

            feedbackArea.classList.add('hidden');

            nextButton.classList.add('hidden');

            learnMoreButton.classList.add('hidden');

            nextButton.textContent = 'Next Question'; // Reset button text for typical flow

            feedbackMessage.textContent = ''; // Clear previous feedback message

        }


        /**

         * Handles the user's answer selection.

         * @param {Event} e - The click event object.

         */

        function selectAnswer(e) {

            if (!quizActive) return; // Prevent interaction if quiz is over


            const selectedButton = e.target;

            const isCorrect = selectedButton.dataset.correct === 'true';


            // Disable all answer buttons after selection

            Array.from(answerButtons.children).forEach(button => {

                button.classList.add('pointer-events-none'); // Disable clicks

                button.classList.remove('selected'); // Remove any lingering selected states

            });


            // Mark selected button

            selectedButton.classList.add('selected');


            // Apply styling based on correctness and update score

            if (isCorrect) {

                selectedButton.classList.add('correct');

                score++;

                feedbackMessage.textContent = "Correct! " + questions[currentQuestionIndex].explanation;

            } else {

                selectedButton.classList.add('incorrect');

                // Find and highlight the correct answer

                Array.from(answerButtons.children).forEach(button => {

                    if (button.dataset.correct === 'true') {

                        button.classList.add('correct');

                    }

                });

                feedbackMessage.textContent = "Incorrect. " + questions[currentQuestionIndex].explanation;

            }


            feedbackArea.classList.remove('hidden'); // Show feedback

            nextButton.classList.remove('hidden'); // Show next button

            learnMoreButton.classList.remove('hidden'); // Show learn more button


            // If it's the last question, change next button text to 'Show Results'

            if (currentQuestionIndex === questions.length - 1) {

                nextButton.textContent = 'Show Results';

            }

        }


        /**

         * Advances to the next question or shows the final results.

         */

        function handleNextButton() {

            currentQuestionIndex++;

            if (currentQuestionIndex < questions.length) {

                showQuestion(); // Load next question

            } else {

                showResults(); // Quiz finished, show results

            }

        }


        /**

         * Displays the final score and offers a restart option.

         */

        function showResults() {

            quizActive = false; // Deactivate quiz interaction

            resetState(); // Clear questions and buttons

            questionDisplay.textContent = 'Quiz Completed!';

            nextButton.classList.add('hidden'); // Hide next button

            learnMoreButton.classList.add('hidden'); // Hide learn more button in results


            scoreDisplay.textContent = `You scored ${score} out of ${questions.length}!`;

            resultsArea.classList.remove('hidden'); // Show results container

        }


        /**

         * Calls the Gemini API to generate more information about the current question's topic.

         */

        async function fetchLearnMore() {

            const currentTopic = questions[currentQuestionIndex].question.split("'")[1] || questions[currentQuestionIndex].question; // Simple way to extract a movie title or topic

            learnMoreButton.textContent = "Loading... ✨";

            learnMoreButton.classList.add('loading');

            learnMoreButton.disabled = true;


            try {

                let chatHistory = [];

                const prompt = `Provide a very brief (2-3 sentences, max 50 words) interesting fact or summary about "${currentTopic}". Do not include the question or answer in your response. Focus on a single interesting aspect.`;

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });

                const payload = { contents: chatHistory };

                const apiKey = ""; // Canvas will provide this at runtime

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });

                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const text = result.candidates[0].content.parts[0].text;

                    // Append the new info to the existing feedback message

                    feedbackMessage.textContent += "\n\n✨ More Info: " + text;

                } else {

                    feedbackMessage.textContent += "\n\nCould not fetch more information at this time.";

                }

            } catch (error) {

                console.error("Error fetching 'learn more' info:", error);

                feedbackMessage.textContent += "\n\nError fetching more information.";

            } finally {

                learnMoreButton.textContent = "Learn More ✨";

                learnMoreButton.classList.remove('loading');

                learnMoreButton.disabled = false;

            }

        }


        /**

         * Calls the Gemini API to generate a new quiz question and adds it to the quiz.

         */

        async function fetchNewQuestion() {

            generateNewQuestionButton.textContent = "Generating... ✨";

            generateNewQuestionButton.classList.add('loading');

            generateNewQuestionButton.disabled = true;


            try {

                let chatHistory = [];

                const prompt = "Generate a challenging movie trivia question with four multiple-choice answers and a brief explanation. Ensure exactly one answer is correct. Provide the output in JSON format.";

                chatHistory.push({ role: "user", parts: [{ text: prompt }] });


                const payload = {

                    contents: chatHistory,

                    generationConfig: {

                        responseMimeType: "application/json",

                        responseSchema: {

                            type: "OBJECT",

                            properties: {

                                "question": { "type": "STRING" },

                                "answers": {

                                    "type": "ARRAY",

                                    "items": {

                                        "type": "OBJECT",

                                        "properties": {

                                            "text": { "type": "STRING" },

                                            "correct": { "type": "BOOLEAN" }

                                        },

                                        "required": ["text", "correct"]

                                    }

                                },

                                "explanation": { "type": "STRING" }

                            },

                            "required": ["question", "answers", "explanation"]

                        }

                    }

                };


                const apiKey = ""; // Canvas will provide this at runtime

                const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${apiKey}`;


                const response = await fetch(apiUrl, {

                    method: 'POST',

                    headers: { 'Content-Type': 'application/json' },

                    body: JSON.stringify(payload)

                });


                const result = await response.json();


                if (result.candidates && result.candidates.length > 0 &&

                    result.candidates[0].content && result.candidates[0].content.parts &&

                    result.candidates[0].content.parts.length > 0) {

                    const jsonString = result.candidates[0].content.parts[0].text;

                    const newQuestion = JSON.parse(jsonString);


                    // Add the new question to the questions array

                    questions.push(newQuestion);

                    // Start the quiz over, which will now include the new question

                    startQuiz();

                } else {

                    // Handle cases where the response structure is unexpected or content is missing

                    alert("Failed to generate a new question. Please try again."); // Using alert only for error display as per strict instruction

                    console.error("Gemini API response structure unexpected:", result);

                }

            } catch (error) {

                console.error("Error generating new question:", error);

                alert("An error occurred while generating a new question."); // Using alert only for error display as per strict instruction

            } finally {

                generateNewQuestionButton.textContent = "Generate New Question ✨";

                generateNewQuestionButton.classList.remove('loading');

                generateNewQuestionButton.disabled = false;

            }

        }



        // Event listeners for navigation and new Gemini API features

        nextButton.addEventListener('click', handleNextButton);

        restartButton.addEventListener('click', startQuiz);

        learnMoreButton.addEventListener('click', fetchLearnMore);

        generateNewQuestionButton.addEventListener('click', fetchNewQuestion);



        // Initial call to start the quiz when the page loads

        document.addEventListener('DOMContentLoaded', startQuiz);

    </script>
